# TODO: Apply Label Encoding to categorical columns

# Create copies to avoid modifying original data
X_train_encoded = X_train.copy()
X_test_encoded = X_test.copy()

# Dictionary to store label encoders for each column
label_encoders = {}

for col in categorical_cols:
  le = LabelEncoder()

  # Fit on training data
  train_vals = X_train_encoded[col].astype(str)

  le.fit(pd.concat([train_vals, pd.Series(["UNK"])], ignore_index=True))

  # Transform train and test
  X_train_encoded[col] = le.transform(X_train_encoded[col].astype(str))

  # Handle unseen categories in test set
  test_vals = X_test_encoded[col].astype(str)
  test_vals = test_vals.apply(lambda x: x if x in le.classes_ else "UNK")
  X_test_encoded[col] = le.transform(test_vals)

  # Store encoder
  label_encoders[col] = le

# Print the shape after encoding
print(f"X_train_encoded shape: {X_train_encoded.shape}")
print(f"X_test_encoded shape: {X_test_encoded.shape}")

def evaluate_model(y_true, y_pred, model_name):
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, average='weighted')
    recall = recall_score(y_true, y_pred, average='weighted')
    f1 = f1_score(y_true, y_pred, average='weighted')

    return {
        'Model': model_name,
        'Accuracy': accuracy,
        'Precision': precision,
        'Recall': recall,
        'F1-Score': f1
    }

TODO: Briefly comment on the strengths and weaknesses of each model, and any impact of class imbalance handling.

  • Logistic Regression:

    • Strengths: It is computationally efficient and outputs interpretable class probabilities. It is less prone to overfitting when the dataset has a low signal-to-noise ratio.
    • Weaknesses: It assumes a linear decision boundary between features and the log-odds of the target, meaning it struggles to capture complex, non-linear feature interactions without manual polynomial expansion.

  • Decision Tree:

    • Strengths: It is a non-parametric model capable of naturally capturing highly non-linear relationships and complex interactions between features through orthogonal splits.
    • Weaknesses: It exhibits high variance and is highly sensitive to training data variations, making it extremely prone to overfitting if tree depth is not restricted.

  • Random Forest:

    • Strengths: It reduces the variance of individual decision trees through bootstrap aggregating (bagging) and random feature selection, yielding higher predictive accuracy and better generalization.
    • Weaknesses: It is computationally more expensive during both training and inference. Additionally, it operates as an ensemble "black box," sacrificing the structural interpretability of a single decision tree.

    Class imbalance biases the model toward the majority class (e.g., heavily predicting "Credit Card" because it appears most frequently), which artificially inflates the Accuracy metric while hiding poor performance on minority classes. Relying on the confusion matrix and metrics like weighted F1-score, Precision, and Recall is essential, as they mathematically penalize the model for misclassifying minority instances and provide a realistic evaluation of its true predictive capability.
