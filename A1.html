# ANSWER: Handle missing values
df_orders_full = df_orders_full.dropna(subset=["order_id"])

num_cols = df_orders_full.select_dtypes(include=["number"]).columns
cat_cols = df_orders_full.select_dtypes(exclude=["number"]).columns

for col in num_cols:
    df_orders_full[col] = df_orders_full[col].fillna(df_orders_full[col].median())
for col in cat_cols:
    if not df_orders_full[col].mode().empty:
        df_orders_full[col] = df_orders_full[col].fillna(df_orders_full[col].mode().iloc[0])

df_orders_full = df_orders_full.dropna(subset=["payment_type"])
